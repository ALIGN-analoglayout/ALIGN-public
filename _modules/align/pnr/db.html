        def init_fn(self, d):
            for (nm,vv) in v:
                def f( x, vv):
                    if vv is None:
# might be faster if we don't do the copy
                        return copy.deepcopy( x)
                    else:
                        klass = globals()[vv]
                        return klass(x)

                if isinstance( vv, tuple):
                    assert vv[0] is list
                    setattr( self, nm, [ f(x,vv[1]) for x in d[nm]])
                else:
                    setattr( self, nm, f(d[nm],vv))
        return init_fn
