[docs]    def parse_args(self, *args, **kwargs):
        arguments = self.parser.parse_args(*args, **kwargs)
        try:
            return schematic2layout(**vars(arguments))
        except Exception:
            logger.exception("Fatal Error. Cannot proceed")
